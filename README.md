### PROGRAM:
#Program to find the distance between two points.
#Developed by: Mohamed Sulthan A

import math
d1=[4,2]
d2=[10,6]
d=math.sqrt((d2[0]-d1[0])**2+(d2[1]-d1[1])**2)
print("{:.2f}".format(d))
